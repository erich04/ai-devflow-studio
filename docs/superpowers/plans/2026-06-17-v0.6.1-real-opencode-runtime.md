# DevFlow Studio v0.6.1 Real opencode Runtime Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Connect the real `opencode serve` HTTP transport behind DevFlow's existing Coding Runtime seam while keeping default verification deterministic on the fake engine.

**Architecture:** DevFlow remains the primary UI and orchestration layer. Electron main owns the opencode process, runtime credentials, permission relay, managed worktree, dependency bootstrap, diff capture, test evidence, and redacted team summary; opencode owns coding execution only.

**Tech Stack:** Electron main process, TypeScript, Vitest, Playwright Electron smoke, existing `CodingEngineAdapter`, existing `opencode-http-adapter.ts`, SQLite local store, Node `child_process.spawn`, `corepack pnpm`.

```ts
import { spawn, type ChildProcess } from 'node:child_process'
import { buildOpencodeServeArgs, listOpencodePermissions } from './opencode-http-adapter.js'

export type ManagedOpencodeServer = {
  projectId: string
  baseUrl: string
  child: ChildProcess
}

export type OpencodeProcessManagerDeps = {
  spawnProcess?: typeof spawn
  findPort?: () => Promise<number>
  waitUntilReady?: (baseUrl: string) => Promise<void>
}

export function createOpencodeProcessManager(deps: OpencodeProcessManagerDeps = {}) {
  const servers = new Map<string, ManagedOpencodeServer>()
  const spawnProcess = deps.spawnProcess ?? spawn
  const findPort = deps.findPort ?? randomLocalPort
  const waitUntilReady = deps.waitUntilReady ?? waitForOpencodeReady

  return {
    async ensure(input: { projectId: string; binaryPath: string; env: NodeJS.ProcessEnv }) {
      const existing = servers.get(input.projectId)
      if (existing && existing.child.exitCode === null) {
        return existing
      }

      const port = await findPort()
      const baseUrl = `http://127.0.0.1:${port}`
      const child = spawnProcess(input.binaryPath, buildOpencodeServeArgs({ hostname: '127.0.0.1', port }), {
        env: input.env,
        stdio: ['ignore', 'pipe', 'pipe'],
      })
      const server: ManagedOpencodeServer = { projectId: input.projectId, baseUrl, child }
      servers.set(input.projectId, server)
      await waitUntilReady(baseUrl)
      return server
    },

    async stopAll() {
      for (const server of servers.values()) {
        server.child.kill()
      }
      servers.clear()
    },
  }
}

async function randomLocalPort(): Promise<number> {
  const { createServer } = await import('node:net')
  return new Promise((resolve, reject) => {
    const server = createServer()
    server.listen(0, '127.0.0.1', () => {
      const address = server.address()
      server.close(() => {
        if (address && typeof address === 'object') resolve(address.port)
        else reject(new Error('Unable to allocate local opencode port'))
      })
    })
    server.on('error', reject)
  })
}

async function waitForOpencodeReady(baseUrl: string) {
  for (let attempt = 0; attempt < 60; attempt += 1) {
    try {
      await listOpencodePermissions({ baseUrl })
      return
    } catch {
      await new Promise((resolve) => setTimeout(resolve, 500))
    }
  }
  throw new Error(`Timed out waiting for opencode serve at ${baseUrl}`)
}
```

```ts
export function createOpencodeHttpCodingEngineAdapter(config: OpencodeHttpCodingEngineConfig): CodingEngineAdapter {
  const sessions = new Map<string, OpencodeRuntimeSession>()
  return {
    engine: 'opencode-http',
    async ensure(input) {
      await config.processManager.ensure({ projectId: input.project.id, binaryPath: config.binaryPath, env: process.env })
      return { projectId: input.project.id, engine: 'opencode-http', status: 'ready' }
    },
    async start(input) {
      const server = await config.processManager.ensure({
        projectId: input.project.id,
        binaryPath: config.binaryPath,
        env: process.env,
      })
      const brief = buildCodingBrief({
        run: input.run,
        node: input.node,
        project: input.project,
        upstreamArtifacts: input.upstreamArtifacts,
        knowledgeReferences: input.knowledgeReferences,
        governanceChecks: input.governanceChecks,
        gateDecisions: input.gateDecisions,
        testEvidence: input.testEvidence,
        userInstruction: input.userInstruction,
        worktreePath: input.workspace.worktreePath,
        branchName: input.workspace.branchName,
      })
      const session = await createOpencodeSession({
        baseUrl: server.baseUrl,
        directory: input.workspace.worktreePath,
        title: `DevFlow ${input.run.title}`,
        model: { providerID: config.providerID, id: config.modelID },
        fetcher: config.fetcher,
      })
      const messagePromise = sendOpencodeMessage({
        baseUrl: server.baseUrl,
        sessionId: session.id,
        model: { providerID: config.providerID, modelID: config.modelID },
        text: brief.prompt,
        fetcher: config.fetcher,
      })
      sessions.set(input.id, { baseUrl: server.baseUrl, sessionId: session.id, directory: input.workspace.worktreePath, messagePromise })
      const permission = await waitForPermission({ baseUrl: server.baseUrl, sessionId: session.id, fetcher: config.fetcher })
      return createOpencodeStartResult(input, brief, permission)
    },
    async approvePermission(input) {
      return approveOpencodePermission({ input, sessions, config })
    },
    async cancel(input) {
      const session = sessions.get(input.codingRun.id)
      if (session) {
        await abortOpencodeSession({
          baseUrl: session.baseUrl,
          sessionId: session.sessionId,
          directory: session.directory,
          fetcher: config.fetcher,
        })
      }
    },
  }
}
```

```ts
import { existsSync } from 'node:fs'
import path from 'node:path'
import { selectDependencyBootstrap, type DependencyBootstrapEvidence } from '@ai-devflow/shared'
import { readProjectFileSnapshot, type LocalTestCommandResult } from './test-runner.js'

export async function runDependencyBootstrap(input: BootstrapRunnerInput): Promise<DependencyBootstrapEvidence> {
  const files = await readProjectFileSnapshot(input.worktreePath)
  const decision = selectDependencyBootstrap({
    files,
    nodeModulesPresent: existsSync(path.join(input.worktreePath, 'node_modules')),
    previousDependencyHash: input.previousDependencyHash,
  })

  if (decision.status === 'skipped' || decision.status === 'needs_approval') {
    return evidenceFromDecision(input, decision, decision.status, null, 0, '', '', decision.reason, true)
  }

  const result = await input.runCommand({
    command: decision.command,
    cwd: input.worktreePath,
    timeoutMs: input.timeoutMs,
  })
  return evidenceFromDecision(
    input,
    decision,
    result.status === 'passed' ? 'passed' : result.status,
    result.exitCode,
    result.durationMs,
    result.stdout,
    result.stderr,
    result.summary,
    result.redacted,
  )
}
```
